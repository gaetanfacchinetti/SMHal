101// defines the new operator (), so that we can access the elements
102// by A(i,j), index going from i=0,...,dim()-1
103double & band_matrix::operator () (int i, int j) {
104 int k=j-i; // what band is the entry
105 assert( (i>=0) && (i<dim()) && (j>=0) && (j<dim()) );
106 assert( (-num_lower()<=k) && (k<=num_upper()) );
107 // k=0 -> diogonal, k<0 lower left part, k>0 upper right part
108 if(k>=0) return m_upper[k][i];
109 else return m_lower[-k][i];
110}

129// LR-Decomposition of a band matrix
130void band_matrix::lu_decompose() {
131 int i_max,j_max;
132 int j_min;
133 double x;
134
135 // preconditioning
136 // normalize column i so that a_ii=1
137 for(int i=0; i<this->dim(); i++) {
138 assert(this->operator()(i,i)!=0.0);
139 this->saved_diag(i)=1.0/this->operator()(i,i);
140 j_min=std::max(0,i-this->num_lower());
141 j_max=std::min(this->dim()-1,i+this->num_upper());
142 for(int j=j_min; j<=j_max; j++) {
143 this->operator()(i,j) *= this->saved_diag(i);
144 }
145 this->operator()(i,i)=1.0; // prevents rounding errors
146 }
147
148 // Gauss LR-Decomposition
149 for(int k=0; k<this->dim(); k++) {
150 i_max=std::min(this->dim()-1,k+this->num_lower()); // num_lower not a mistake!
151 for(int i=k+1; i<=i_max; i++) {
152 assert(this->operator()(k,k)!=0.0);
153 x=-this->operator()(i,k)/this->operator()(k,k);
154 this->operator()(i,k)=-x; // assembly part of L
155 j_max=std::min(this->dim()-1,k+this->num_upper());
156 for(int j=k+1; j<=j_max; j++) {
157 // assembly part of R
158 this->operator()(i,j)=this->operator()(i,j)+x*this->operator()(k,j);
159 }
160 }
161 }
162}
163// solves Ly=b
164std::vector<double> band_matrix::l_solve(const std::vector<double>& b) const {
165 assert( this->dim()==(int)b.size() );
166 std::vector<double> x(this->dim());
167 int j_start;
168 double sum;
169 for(int i=0; i<this->dim(); i++) {
170 sum=0;
171 j_start=std::max(0,i-this->num_lower());
172 for(int j=j_start; j<i; j++) sum += this->operator()(i,j)*x[j];
173 x[i]=(b[i]*this->saved_diag(i)) - sum;
174 }
175 return x;
176}
177// solves Rx=y
178std::vector<double> band_matrix::r_solve(const std::vector<double>& b) const {
179 assert( this->dim()==(int)b.size() );
180 std::vector<double> x(this->dim());
181 int j_stop;
182 double sum;
183 for(int i=this->dim()-1; i>=0; i--) {
184 sum=0;
185 j_stop=std::min(this->dim()-1,i+this->num_upper());
186 for(int j=i+1; j<=j_stop; j++) sum += this->operator()(i,j)*x[j];
187 x[i]=( b[i] - sum ) / this->operator()(i,i);
188 }
189 return x;
190}
191
192std::vector<double> band_matrix::lu_solve(const std::vector<double>& b,
193 bool is_lu_decomposed) {
194 assert( this->dim()==(int)b.size() );
195 std::vector<double> x,y;
196 if(is_lu_decomposed==false) {
197 this->lu_decompose();
198 }
199 y=this->l_solve(b);
200 x=this->r_solve(y);
201 return x;
202}

211void spline::set_points(const std::vector<double>& x,
212 const std::vector<double>& y, bool cubic_spline) {
213 assert(x.size()==y.size());
214 m_x=x;
215 m_y=y;
216 int n=x.size();
217 // TODO sort x and y, rather than returning an error
218 for(int i=0; i<n-1; i++) {
219 assert(m_x[i]<m_x[i+1]);
220 }
221
222 if(cubic_spline==true) { // cubic spline interpolation
223 // setting up the matrix and right hand side of the equation system
224 // for the parameters b[]
225 band_matrix A(n,1,1);
226 std::vector<double> rhs(n);
227 for(int i=1; i<n-1; i++) {
228 A(i,i-1)=1.0/3.0*(x[i]-x[i-1]);
229 A(i,i)=2.0/3.0*(x[i+1]-x[i-1]);
230 A(i,i+1)=1.0/3.0*(x[i+1]-x[i]);
231 rhs[i]=(y[i+1]-y[i])/(x[i+1]-x[i]) - (y[i]-y[i-1])/(x[i]-x[i-1]);
232 }
233 // boundary conditions, zero curvature b[0]=b[n-1]=0
234 A(0,0)=2.0;
235 A(0,1)=0.0;
236 rhs[0]=0.0;
237 A(n-1,n-1)=2.0;
238 A(n-1,n-2)=0.0;
239 rhs[n-1]=0.0;
240
241 // solve the equation system to obtain the parameters b[]
242 m_b=A.lu_solve(rhs);
243
244 // calculate parameters a[] and c[] based on b[]
245 m_a.resize(n);
246 m_c.resize(n);
247 for(int i=0; i<n-1; i++) {
248 m_a[i]=1.0/3.0*(m_b[i+1]-m_b[i])/(x[i+1]-x[i]);
249 m_c[i]=(y[i+1]-y[i])/(x[i+1]-x[i])
250 - 1.0/3.0*(2.0*m_b[i]+m_b[i+1])*(x[i+1]-x[i]);
251 }
252 } else { // linear interpolation
253 m_a.resize(n);
254 m_b.resize(n);
255 m_c.resize(n);
256 for(int i=0; i<n-1; i++) {
257 m_a[i]=0.0;
258 m_b[i]=0.0;
259 m_c[i]=(m_y[i+1]-m_y[i])/(m_x[i+1]-m_x[i]);
260 }
261 }
262
263 // for the right boundary we define
264 // f_{n-1}(x) = b*(x-x_{n-1})^2 + c*(x-x_{n-1}) + y_{n-1}
265 double h=x[n-1]-x[n-2];
266 // m_b[n-1] is determined by the boundary condition
267 m_a[n-1]=0.0;
268 m_c[n-1]=3.0*m_a[n-2]*h*h+2.0*m_b[n-2]*h+m_c[n-2]; // = f'_{n-2}(x_{n-1})
269}
270
271double spline::operator() (double x) const {
272 size_t n=m_x.size();
273 // find the closest point m_x[idx] < x, idx=0 even if x<m_x[0]
274 std::vector<double>::const_iterator it;
275 it=std::lower_bound(m_x.begin(),m_x.end(),x);
276 int idx=std::max( int(it-m_x.begin())-1, 0);
277
278 double h=x-m_x[idx];
279 double interpol;
280 if(x<m_x[0]) {
281 // extrapolation to the left
282 interpol=((m_b[0])*h + m_c[0])*h + m_y[0];
283 } else if(x>m_x[n-1]) {
284 // extrapolation to the right
285 interpol=((m_b[n-1])*h + m_c[n-1])*h + m_y[n-1];
286 } else {
287 // interpolation
288 interpol=((m_a[idx]*h + m_b[idx])*h + m_c[idx])*h + m_y[idx];
289 }
290 return interpol;
291}
